import { Department } from './../../models/Department';
import { DepartmentService } from './../../services/department.service';
import { Component, Input, OnInit } from '@angular/core';
import { Observable } from 'rxjs';

@Component({
  selector: 'app-departments',
  templateUrl: './departments.component.html',
  styleUrls: ['./departments.component.scss']
})
export class DepartmentsComponent implements OnInit {
  departmentsCount: number;

  constructor(private departmentService : DepartmentService) { }

  departmentList$: Observable<Department[]>;
  department$: Observable<Department>;

  @Input() department:any;
  id: number = 0;
  name: string = "";
  modalTitle: string ="";

  ngOnInit() {
    this.departmentList$ = this.departmentService.getDepartments();
    this.name = this.department.name;
    this.departmentsCount = this.departmentService.depNumber;
  }

  updateUser() {
    throw new Error('Method not implemented.');
  }

  activateDepartmentComponent: boolean = false;

  deleteDepartment(department:any) {
    if(confirm(`Tem certeza que deseja excluir o departamento ${department.name}?`)) {
      this.departmentService.deleteDepartment(department.id).subscribe(res => {
        var closeModalBtn = document.getElementById('edit-modal');
      if(closeModalBtn) {
        closeModalBtn.click();
      }

      this.departmentList$ = this.departmentService.getDepartments();
      })
    }
  }

  editDepartment(id: any, department: any) {
    var department: any = {
      id: this.id,
      name: this.name
    }

    this.departmentService.updateDepartment(id, department).subscribe(res => {
      var closeModalBtn = document.getElementById('edit-modal');
      if(closeModalBtn) {
        closeModalBtn.click();
      }
    })
  }

  modalEdit(item:any) {
    this.department = item;
    this.modalTitle = "Edit Inspection";
    this.activateDepartmentComponent = true;
  }

  modalClose() {
    this.activateDepartmentComponent = false;
    this.departmentList$ = this.departmentService.getDepartments();
  }
}
